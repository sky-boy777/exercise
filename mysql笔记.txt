创建
create database 数据库名  dufault charset='utf8';

删除
drop database 数据库名;
drop table  表名;

删除表数据
delete 表名 where 条件;

插入
insert into 表名(字段名) values(值);
insert into article_type(type_name) values('python');
--------------------------------------------------------------------------------------------------
导出：mysqldump -uroot -p 数据库名 >  文件名.sql
导入： mysql -uroot -p  数据库名（提前创建）<  文件名.sql

更新：update 表名 set 列名=新值 where 列名=条件值;
更改字段类型：alter table 表名 modify column 字段名  类型  varchar(300);
